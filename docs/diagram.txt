  id int [pk]
  agency_id int
  status_id int
  campaign_id int
}

Ref: orders.agency_id < agencies.id
Ref: orders.status_id < statuses.id
Ref: orders.campaign_id < campaigns.id

  id int [pk]
  title string
 }

Table statuses {
  id int [pk]
  title string
}
 
Table campaigns {
  id int [pk]
  unit_number int
  street_number int
  street_name string
  suburb_name string
}
